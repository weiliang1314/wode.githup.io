矩形重叠

问题：矩形以列表[x1,y1,x2,y2],其中（x1,y1）为左下角的坐标，（x2,y2）为右上角的坐标，如果两个矩形相交的面积为正
则两个矩形重叠，只在角或边接触的两个矩形不构成重叠,判断是否有相交的面积，返回布尔值
问题分析：根据问题当其中一个矩形在另一个矩形四周时它们不重叠，判断在四周的四种情况：rec1右<rec2左，rec1左>rec2右，rec1下>rec2上，rec1上<rec2下

代码实现

let isRectanglevoverlap = function(rec1, rec2) {
  if (rec1[2] <= rec2[0] || rec1[0] >= rec2[2] || rec1[1] >= rec2[3] || rec1[3] <= rec2[1]) {
   return false
 } else {
   return true
 }

 }

矩阵置为零

问题：二维数组由0，1组成，把有0的整行，整列都变为0
问题分析：首先我们要考虑第一行，第一列是否有0，若有0则就要把其所在行或列都置为0；判断其余行列是否有0，若有则要把其所在行和列的第一个都置为0，再把第一行和第一列有0的都置为0

代码实现

 let setZero = function(matrix) {
//检查第一行，第一列是否有0，
let firstCol = false; //列
let firstRow = false;
for (let i = 0; i < matrix.length; i++) {
    if (matrix[i][0] === 0) {
        firstCol = true
    }
   }
   for (let i = 0; i < matrix[0].length; i++) {
    if (matrix[0][i] === 0) {
        firstRow = true
    }
     }
   //第一行与第一列标记其余行列的0
      for (let row = 1; row < matrix.length; row++) {
    for (let col = 1; col < matrix[0].length; col++) {
        if (matrix[row][col] === 0) {
            matrix[row][0] = 0;
            matrix[0][col] = 0;

        }
         }
       }
      for (let row = 1; row < matrix.length; row++) {
      for (let col = 1; col < matrix[0].length; col++) {
        if (matrix[row][0] === 0 || matrix[0][col] === 0) {
            matrix[row][col] = 0;


        }
       }
     }
    //处理第一行，第一列有0的情况
     if (firstCol === true) {
    for (let i = 0; i < matrix.length; i++) {
        matrix[i][0] = 0;
    }
    }
     if (firstRow === true) {
    for (let i = 0; i < matrix[0].length; i++) {
        matrix[0][i] = 0;
    }
   }
    return matrix;

  }
