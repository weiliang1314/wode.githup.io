防抖函数

描述：当持续触发事件时，n秒内没有再触发事件，事件处理函数才会执行一次，如果设定的时间到来之前，又一次触发了事件，就重新开始延时。
用途：搜索联想，用户在不断输入值时，用防抖来节约请求资源。

代码实现

let input = document.querySelector('input');

function debounce(func, delay) {
let timer = null;

 return function(...args) {
    if (timer) clearTimeout(timer);

    timer = setTimeout(() => {
        //有参数时
        func.apply(this, args); //func();
           }, delay)
        }
   }

节流函数

描述：连续触发事件但是在n秒中只执行一次函数。
用途：鼠标连续不断地触发某事件，只在单位时间内只触发一次；在页面的无限加载场景下，需要用户在滚动页面时，每隔一段时间发一次请求，而不是在用户停下滚动页面操作时才去请求数据。

代码实现

function throttle(fn, delay) {
let timer;
return function(...args) {
    if (timer) {
        return;
       }
    timer = setTimeout(() => {
        // fn.apply(this, arguments);或
        //三点运算符，代替arguments:传递给函数的隐式参数，调用传递给函数fn实参
        fn(...args);

        timer = null;
           }, delay)
         }

     }
