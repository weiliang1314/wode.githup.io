tags: 冒泡排序改进，找到一个枢纽左边放小于它的右边放大于它的（多次）

代码封装

//选择枢纽
      arraylist.prototype.median = function(left, right) {
              var center = Math.floor((left + right) / 2)
                  //判断大小进行交换
              if (this.array[left] > this.array[center]) {
                  this.swap(left, center)
              } else if (this.array[center] > this.array[right]) {
                  this.swap(center, right)
              }
              this.swap(center, right - 1)
              return this.array[right - 1]

          }
          //快速排序
      arraylist.prototype.quicks = function() {
          this.quick(0, this.array.length - 1)
      }
      arraylist.prototype.quick = function(left, right) {
          //结束条件
          if (left >= right) return
              //获取枢纽
          var pivot = this.median(left, right)
              //定义变量，记录当前找到的位置
          var i = left
          var j = right - 1
              //进行交换
          while (true) {
              while (this.array[++i] < pivot) {}
              while (this.array[--j] > pivot) {}
              if (i < j) {
                  this.swap(i, j)

              } else {
                  break
              }
          }
          //将枢纽放在正确的位置
          this.swap(i, right - 1)
              //递归
          this.quick(left, i - 1)
          this.quick(i + 1, right)
      }
  }
  //测试
  var list = new arraylist()

list.prototype.shell = function() {
             var length = this.array.length
                 //初始化间隔增量
             var geg = Math.floor(length / 2)
                 //geg不断减小
             while (geg >= 1) {
                 for (var i = geg; i < length; i++) {
                     var temp = this.array[i]
                     var j = i
                     while (this.array[j - geg] > temp && j > geg - 1) {
                         this.array[j] = this.array[j - geg]
                         j -= geg
                     }
                     this.array[j] = temp
                 }
                 geg = Math.floor(geg / 2)

             }
         }

javascript中for的作用域

title: ‘’javascript中for的作用域’’
date: 2020-05-23 12:16:41
tags: 在ES5前if与for没有块级作用域的概念，在ES6中添加let这一定义变量的关键字，使其在块级作用域有效

使用var定义变量

<button>按钮1</button>
<button>按钮2</button>
<button>按钮3</button>
<button>按钮4</button>
<button>按钮5</button>
<script>
因为没有块级作用域，按钮点击后函数执行当i找到是i的值已经被改变
var btn = document.getElementsByTagName("button");
                  for (var i = 0; i < btn.length; i++) {
                      btn[i].addEventListener('click', function() {
                          console.log("第" + i + "个按钮点击")
                      })</script>

使用闭包解决

当匿名函数执行时，函数中有块级作用域，每次i就会被传入，但外面i的改变不会影响函数里面的i
var btn = document.getElementsByTagName("button");
     for (var i = 0; i < btn.length; i++) {
         (function(num) {
             btn[i].addEventListener('click', function() {
                 console.log("第" + num + "个按钮点击")
             })
         })(i)

使用ES6中let解决

const btn = document.getElementsByTagName("button");
  for (let i = 0; i < btn.length; i++) {
      btn[i].addEventListener('click', function() {
          console.log("第" + i + "个按钮点击")
      })
  }

        if (newnode.key < node.key) {
            //向左查找
            if (node.left == null) {
                node.left = newnode
            } else { //递归

            }
