javascript中for的作用域

title: ‘’javascript中for的作用域’’
date: 2020-05-23 12:16:41
tags: 在ES5前if与for没有块级作用域的概念，在ES6中添加let这一定义变量的关键字，使其在块级作用域有效

使用var定义变量

<button>按钮1</button>
<button>按钮2</button>
<button>按钮3</button>
<button>按钮4</button>
<button>按钮5</button>
<script>
因为没有块级作用域，按钮点击后函数执行当i找到是i的值已经被改变
var btn = document.getElementsByTagName("button");
                  for (var i = 0; i < btn.length; i++) {
                      btn[i].addEventListener('click', function() {
                          console.log("第" + i + "个按钮点击")
                      })</script>

使用闭包解决

当匿名函数执行时，函数中有块级作用域，每次i就会被传入，但外面i的改变不会影响函数里面的i
var btn = document.getElementsByTagName("button");
     for (var i = 0; i < btn.length; i++) {
         (function(num) {
             btn[i].addEventListener('click', function() {
                 console.log("第" + num + "个按钮点击")
             })
         })(i)

使用ES6中let解决

const btn = document.getElementsByTagName("button");
  for (let i = 0; i < btn.length; i++) {
      btn[i].addEventListener('click', function() {
          console.log("第" + i + "个按钮点击")
      })
  }

    binarserach.prototype.insertNode = function(node, newnode) {
        if (newnode.key < node.key) {
            //向左查找
            if (node.left == null) {
                node.left = newnode
            } else { //递归
                this.insertNode(node.left, newnode)
            }
        } else {
            //向右查找
            if (node.right == null) {
                node.right = newnode
            } else {
                this.insertNode(node.right, newnode)
            }
        }
    }
}

tags: 闭包导致的内存溢出

函数执行完后，函数内的局部变量没有释放，占用内存时间变长，容易照成内存泄漏
解决：尽量不用闭包，及时释放
 function fn1() {
     var arr = new Array[1000]

     function fn2() {
         console.log(arr.length)
     }
     return fn2
 }
 var f = fn1()
 f()
     //f=null及时释放
