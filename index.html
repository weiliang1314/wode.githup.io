什么是动态规划算法

动态规划常常适用于有重叠子问题和最优子结构性质的问题，动态规划方法所耗时间往往远少于暴力解法。大致上，若要解一个给定问题，我们需要解其不同部分（即子问题），再合并子问题的解以得出原问题的解。通常许多子问题非常相似，为此动态规划法只解决每个子问题一次，从而减少计算量，一旦某个给定子问题的解已经算出，则将其存储，以便下次需要同一个子问题解之时直接查表。

三要素

  1. 最优子结构：子问题的最优解能够决定这个问题的最优解
  2. 边界：问题最小子集的解
  3. 状态转移函数：递推式

    解决爬楼梯问题

问题：一共有n阶楼梯，每次只能爬一阶或两阶，有多少种爬的方法
分析与解决：在这个问题中我首先考虑到当n=1时f(1)=1,当n=2时f(2)=2,但当n>2时发生了变化，f(3)=3,f(4)=5…
由此得出公式：f(n)=f(n-1)+f(n-2) n>2;f(1)与f(2)相当于边界，公式就是状态转移函数，f(n-1)与f(n-2)就是最优子结构
function pa(n) {
if(n<1){return 0;}
if (n == 1 || n == 2) {
return n;
}
let w = [];
w[0] = 1;
w[1] = 2;
for (let i = 2; i < n; i++) {
w[i] = w[i - 1] + w[i - 2];
}
return w[n - 1];
}

解决最低票价问题

问题：假设旅游时,有三种票价：玩一天costs[0]元，连续玩七天costs[1]元，连续玩三十天costs[2]元，一年365天数组days相当于每天1~365，dyas[]中的天数是有旅游的，自定义函数返回days[]中每一天最低票价。
问题分析：
function mincost(days, costs) {
let dp = new Array(days[days.length - 1] + 1).fill(0); //使用固定值替换数组元素初始化全为0，dp[0]=0没有旅行
let daysIndex = 0;
for (let i = 1; i < dp.length; i++) {
if (i === days[daysIndex]) {
dp[i] = Math.min(
dp[Math.max(0, i - 1)] + costs[0],
//在i-7天买了一张七天票，现在还能用,在i-7这一天只买了还没用
dp[Math.max(0, i - 7)] + costs[1],
dp[Math.max(0, i - 30)] + costs[2],
)
daysIndex++;
} else {
//没有费用
dp[i] = dp[i - 1]
}
}
return parseInt(dp[dp.length - 1]) //完成当天以及之前天，花费最少费用
}

ajax实现基本流程

  1. 创建ajax对象
  2. 告诉请求方式与地址
  3. 发送请求
  4. 获取服务器端给客户端的数据

    封装注意的问题

1、请求参数位置：将请求参数传递到ajax函数内部，在函数内部根据请求方式不同将请求参数的位置放在不同位置
get 放在请求地址后面，post放在send()里面
2、请求参数格式 ：application/x-www-form-urlencoded（默认）,application/json 传递对象数据类型对于函数的调用者更加友好
3、如果请求方式是post，则需要在发送请求之前对请求的参数类型进行设置xhr.setRequestHeader(‘Content-Type’, contentType) json格式，需要把对象转换为json字符串
4、使用Object.assign()对设定默认值进行覆盖

封装代码

function ajax(options) {
     //默认值设置
     let defaults = {
             type: 'get',
             url: '',
             data: {},
             header: {
                 'Content-Type': 'application/x-www-form-urlencoded'
             },
             success: function() {},
             error: function() {}

         }
         //如果使用ajax时没有自定义，就使用默认值，如果使用了options就会覆盖默认值defults
         //对象覆盖方法
     Object.assign(defaults, options)

     //创建ajax对象
     let xhr = new XMLHttpRequest();
     //拼接请求参数的变量
     let params = '';
     for (let attr in defaults.data) {
         //将参数转换成字符串
         params += attr + '=' + defaults.data[attr] + '&'
     }
     //截取字符串最后面&
     params = params.substr(0, params.length - 1)
         //请求方式判断
     if (defaults.type == 'get') {
         defaults.url = defaults.url + '?' + params;
     }
     //配置对象
     xhr.open(defaults.type, defaults.url);
     //发送请求
     if (options.type == 'post') {
         let contentType = defaults.header['Content-Type']
             //请求参数类型设置

         xhr.setRequestHeader('Content-Type', contentType);
         if (contentType == 'application/json') {
             xhr.send(JSON.stringify(defaults.data))
         } else {
             xhr.send(params)
         }
     } else {
         xhr.send();
     }
     //当xhr对象接收完响应数据后触发，服务器端传过来的数据
     xhr.onload = function() {
         //获取响应头的数据
         let content = xhr.getResponseHeader('Content-Type')
             //服务器端返回的数据
         let responseText = xhr.responseText;
         //判断字符串是否包含某个字符
         if (content.includes('application/json')) {
             responseText = JSON.parse(responseText)
         }
         //请求成功http状态码
         if (xhr.status == 200) {
             //请求成功
             options.sucess(responseText, xhr);
         } else {
             //请求失败
             options.error(responseText, xhr)
         }
     }
   }

express介绍

Express 是一个简洁而灵活的 node.js Web应用框架, 提供了一系列强大特性帮助你创建各种 Web 应用，和丰富的 HTTP 工具。
使用 Express 可以快速地搭建一个完整功能的网站。

特性

  • 设置中间件来响应 HTTP 请求。
  • 定义路由表用于执行不同的 HTTP 请求动作。
  • 通过向模板传递参数来动态渲染 HTML 页面。

    快速排序

    title: 快速排序
    date: 2020-09-19 20:21:12
    tags: 冒泡排序改进，找到一个枢纽左边放小于它的右边放大于它的（多次）

    代码封装

    //选择枢纽
      arraylist.prototype.median = function(left, right) {
              var center = Math.floor((left + right) / 2)
                  //判断大小进行交换
              if (this.array[left] > this.array[center]) {
                  this.swap(left, center)
              } else if (this.array[center] > this.array[right]) {
                  this.swap(center, right)
              }
              this.swap(center, right - 1)
              return this.array[right - 1]

          }
          //快速排序
      arraylist.prototype.quicks = function() {
          this.quick(0, this.array.length - 1)
      }
      arraylist.prototype.quick = function(left, right) {
          //结束条件
          if (left >= right) return
              //获取枢纽
          var pivot = this.median(left, right)
              //定义变量，记录当前找到的位置
          var i = left
          var j = right - 1
              //进行交换
          while (true) {
              while (this.array[++i] < pivot) {}
              while (this.array[--j] > pivot) {}
              if (i < j) {
                  this.swap(i, j)

              } else {
                  break
              }
          }
          //将枢纽放在正确的位置
          this.swap(i, right - 1)
              //递归
          this.quick(left, i - 1)
          this.quick(i + 1, right)
      }
  }

  var list = new arraylist()
