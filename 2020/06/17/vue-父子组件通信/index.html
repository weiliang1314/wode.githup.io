vue-父子组件通信

title: vue.js-父子组件通信
date: 2020-06-17 19:22:35
tags: 父传子通过props传递数据，子传父通过自定义事件

父传子

 其中因为组件无法直接访问Vue实例对象，组件有自己存放数据属性data但其必须为函数，且返回一个对象，对象中存放数据
 <div id="app">
    <!--此处不支持驼峰命名，改为此形式-->
    <cpn :c-info='info'></cpn>
</div>
<template id="cpn">
<h1>{{cInfo}}</h1>
</template>
<script src='./vue.js'></script>
<script>
    const cpn = {
        template: '#cpn',
        props: {
            cInfo: {
                type: Object,
                default () {
                    return {}
                }
            }
        },
        data(){
          return{}
        }
    }
    const app = new Vue({
        el: '#app', //挂载要管理的元素
        data: {
            info: {
                name: '我的',
                age: 19,
                sex: '男'
            },
        },
        components: {
            cpn
        }

    })
</script>

子传父

   <!--父组件模板-->
<div id='app'>
    <!--监听子组件发射的事件-->
    <cpn @itemclick='cpnclick'></cpn>
</div>
<!--子组件模板-->
<template id="cpn">
  <div>
<button v-for="item in categories" @click='btn(item)'>{{item.name}}</button></div>
 </template>
<script src='./vue.js'></script>
<script>
    //子组件
    const cpn = {
            template: '#cpn',
            data() {
                return {
                    categories: [{
                        id: 'aaa',
                        name: '热门'
                    }, {
                        id: 'qqq',
                        name: '热搜'
                    }, {
                        id: 'sss',
                        name: '搜狗'
                    }, {
                        id: 'ooo',
                        name: '该带'
                    }, ]
                }
            },
            methods: {
                btn(item) {
                    //console.log(item)
                    //发射事件，自定义事件
                    this.$emit('itemclick', item)
                }
            }

        }
        //父组件
    const app = new Vue({
        el: '#app', //挂载要管理的元素
        data: {
            message: '你电脑你打击违法',
            movies: ['好多', '有点', '大话西游', '少年']
        },
        components: {
            cpn
        },
        methods: {
            cpnclick(item) {
                console.log('cpnclick', item)

            }
        }
    })
</script>
