硬币问题

问题：给定不同面额硬币（每种面额有无数个），再给一个数，让这些硬币凑成这个数，返回凑成这个数的有几种方法
问题分析：从凑0，1，2…开始凑一直到目标数，硬币面额种类从0，1，2…种累加到给定的种类，当我们需要凑0时，不管有几种
面额硬币都不给他d(0)=1;当所要凑得数小于某种面额硬币时那么这种硬币是不能凑成的，当选择一个硬币时凑数时，
当前所要凑得数=原来要凑的数-所选硬币面额

代码实现

   let change = function(amount, coins) {
 let dp = [Array(amount + 1).fill(0)];
 console.log(dp)
 dp[0][0] = 1;
 for (let i = 1; i < coins.length + 1; i++) {
    dp[i] = Array(amount + 1).fill(0);
    for (let j = 0; j < amount + 1; j++) {
        dp[i][j] = dp[i - 1][j] + (j >= coins[i - 1] ? dp[i][j - coins[i - 1]] : 0)
     }
      }
return dp[coins.length][amount];

   }

最大正方形

问题：输入一个0和1组成的二维矩阵，找到只包含1的最大正方形，返回面积
问题分析：当1在边界时，最大正方形面积为1，边长d=1;当所选1的左边，上边，左上边，都不是0，它的
边长才能加1，当前边长=上一次最短边长+1

代码实现

let maxzheng = function(martix) {
let maxSidelength = 0; //记录最大边长

 let dp = new Array(martix.length);
for (let i = 0; i < martix.length; i++) {
    dp[i] = new Array(martix[i].length).fill(0); //把它变成一个二维数组，初始值为0
    for (let j = 0; j < martix[i].length; j++) {
        if (martix[i][j] == '1') {
            if (i == 0 || j == 0) { dp[i][j] = 1; } else {
                dp[i][j] = Math.min(dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]) + 1 //左边，上边，左上
            }
            maxSidelength = Math.max(maxSidelength, dp[i][j])

        }
          }
            }
          return maxSidelength * maxSidelength;
          }
