打家劫舍问题（动态规划）

问题：给定一个数组，代表每个房屋存放的金额，当偷相邻两个房屋时就会报警，计算在不触发报警系统的情况下偷盗的最大金额
问题分析：数组中的值为1个或0个则问题不成立，假设数组中有3个以上数值，所取得数不能相邻，当前索取金额总数=正在取+之前取,所取得金额总数最大;m[i]=n[i]+m[i-2]或m[i]=mi-1

代码实现

  let rob = function(nums) {
    //边界
    if (nums.length === 0) {
        return 0
    }
    if (nums.length === 1) {
        return nums[0]
    }
    const memo = [];
    memo[0] = nums[0];
    memo[1] = Math.max(nums[0], nums[1]);
    for (let i = 2; i <= nums.length; i++) {
        memo[i] = Math.max(nums[i] + memo[i - 2], memo[i - 1]);

    }
    return memo[nums.length - 1];
}

岛屿数量问题（dfs）

深度优先搜索算法：一种用于遍历或搜索树或图的算法。 沿着树的深度遍历树的节点，尽可能深的搜索树的分支。当节点v的所在边都己被探寻过或者在搜寻时结点不满足条件，搜索将回溯到发现节点v的那条边的起始节点。整个进程反复进行直到所有节点都被访问为止。
问题：1为陆地，0为水，二维网格，计算岛屿数量，岛屿只能由水平方向和\或垂直方向上相邻的陆地连接形成，边界外都是水
问题分析：先遍历找到二位数组中为1的点，再对其四个方向进行判断，其中需要注意已经判定过得点置为0；每找到为1的点就进行判断，直到遍历完所有

代码实现

let num = function(grid) {
 let count = 0; //记录数量
   for (let i = 0; i < grid.length; i++) {
  for (let j = 0; j < grid[i].length; j++) {
      if (grid[i][j] == '1') {
          count++;
          turn(i, j, grid)
      }
    }
    }
  return count;
        }

   function turn(i, j, grid) {
  //边界判断
  if (grid[i] == undefined || grid[i][j] == undefined || grid[i][j] == '0') {
  return;
}
grid[i][j] = '0';
turn(i + 1, j, grid)
   turn(i - 1, j, grid)
 turn(i, j + 1, grid)
turn(i, j - 1, grid)
   }
